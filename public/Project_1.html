In [19]:
calories_per_minute = []

for i in range(len(df_cleaned)):
    # Splitting the duration into, hours, minutes, and seconds.
    time_parts = df_cleaned['Duration'].iloc[i].split(':')
    
  # If the duration has 3 parts, it includes hours
    if len(time_parts) == 3:
        total_minutes = int(time_parts[0]) * 60 + int(time_parts[1]) + int(time_parts[2]) / 60
    # If the duration has 2 parts, it's just minutes and seconds
    elif len(time_parts) == 2:
        total_minutes = int(time_parts[0]) + int(time_parts[1]) / 60
    
    # Calculate calories per minute
    if total_minutes > 0:  
        calories_per_minute.append(df_cleaned['Calories Burned'].iloc[i] / total_minutes)
    else:
        calories_per_minute.append(0)

df_cleaned.loc[:, 'calories_per_minute'] = calories_per_minute

df_final = df_cleaned
